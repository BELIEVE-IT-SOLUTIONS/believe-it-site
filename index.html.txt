/* Counter */
let count=0;

let speed=10;
function updateCounter(){

    setTimeout(updateCounter,speed);
  }
}

  entries.forEach(entry=>{
    if(entry.isIntersecting){
      entry.target.classList.add('visible');
    }
  });
});
faders.forEach(f=>observer.observe(f));
